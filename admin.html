Безопасность

Правила Firestore

Вставь в Firebase Console → Firestore Database → Rules → Publish: 

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Каталог предметов — читают все, пишут только авторизованные
    match /subject_catalog/{docId} {
      allow read: if true;
      allow create, update: if request.auth != null
        && request.resource.data.keys().hasAll(['title', 'slug', 'sort_order']);
      allow delete: if request.auth != null;
    }

    // Шаблоны заданий — читают все, пишут только авторизованные
    match /task_templates/{docId} {
      allow read: if true;
      allow create, update: if request.auth != null
        && request.resource.data.keys().hasAll(['catalog_id', 'title', 'order_index']);
      allow delete: if request.auth != null;
    }

    // Ученики — читают все (поиск по токену ?k=TOKEN), пишут только авторизованные
    match /users/{userId} {
      allow read: if true;
      allow create: if request.auth != null
        && request.resource.data.keys().hasAll(['name', 'access_token', 'is_active']);
      allow update, delete: if request.auth != null;
    }

    // Предметы ученика
    match /users/{userId}/subjects/{subjectId} {
      allow read: if true;
      allow create, update: if request.auth != null
        && request.resource.data.keys().hasAll(['catalog_id', 'title']);
      allow delete: if request.auth != null;
    }

    // Задания ученика
    match /users/{userId}/subjects/{subjectId}/tasks/{taskId} {
      allow read: if true;
      allow create, update: if request.auth != null
        && request.resource.data.keys().hasAll(['title', 'status']);
      allow delete: if request.auth != null;
    }
  }
}
